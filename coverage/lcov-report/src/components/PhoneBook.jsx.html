import React from 'react';
 
const PhoneBook = ({names, numbers, getNewName, iptValue, addName, phoneNumber, getNewNumber, filterVal, getfilteredName, findRecord, nameFound}) => {
 
    return(
      <div className="phonebook-container">
      <h1>Phone book</h1>
 
      <div className = "filter">
         Find a Name: <input type="text" value = {filterVal} onChange = {getfilteredName} />
         <button onClick = {findRecord}>find</button>
      </div>
      <div className="found-name">
      <ul>
         
            <li key = {nameFound.name}>{nameFound.name}  {nameFound.number} </li>
         
      </ul>
      </div>
      <h1>Add a new Name</h1>
      
      <form>
         <div>
            name: <input value = {iptValue} onChange = {getNewName}/>
         </div>
         <div>
            number: <input value = {phoneNumber}  onChange = {getNewNumber}/>
         </div>
         <div>
            <button onClick={addName} type = 'submit'>Add</button>
         </div>
      </form>
 
      <h2>Numbers</h2>
      <div className="list-container">
 
      
      <ul>
        {
            names.map(name => {
            return <li key={name.name}>{name.name}: </li>
         })
        }
      </ul>
 
      <ul>
      {
          numbers.map(number => <li key={number.number}>{number.number}</li>)
        }
      </ul>
      </div>
      </div>
    );
}
 
export default PhoneBook;
