import React from 'react';
 
const DisplayCountry = ({countryObj, show, weatherObj}) => {
 
    
    if(countryObj.length > 10) {
        return(
            <p>too many matches, please specify another filter...</p>
        )
    } 
    
    else if(countryObj.length > 1 && countryObj.length < 10) {
        // const [Country, setCountry] = useState({});
 
        return(
            <>
             <ul>
                {
                    countryObj.map((country,i) => {
                    let countries = [];
                    countries.push(country.name.official);
                    return (
                        <>
                        <div style = {{display: 'flex'}}>
                         <li key = {i}>{country.name.official}</li>
                         <button id = {i} onClick = {show}>show</button>
                         </div>
                        </>
                    )
                    
                 })
                }
             </ul>
            </>
        )
    } else if(countryObj.length === 1) {
        return(
            <>
             {countryObj.map((country) =>{
                let languages = Object.values(country.languages);
                return (
                    <>  
                    <h1>{country.name.official}</h1>
 
                    <h3>capital: {country.capital}</h3>
                    <h3>Area: {country.area}</h3>
 
                    <h4>laguages:</h4>
                    <ul>
                        {
                          languages.map((language, i) => <li key = {i}>{language}</li>)
                        }
                    </ul>
 
                    <img src={country.flags.png} alt="nation flag" />
 
                    <h3>weather in {country.name.official}</h3>
 
                     
                    </>
                )
             })
             
             
             }
 
             {
                
                    weatherObj.map(weather => {
                    return(
                        <>
                         <h3>condition: {weather.condition}</h3>
                         <h3>temperature: {weather.temp}Â°C</h3>
                         <img src={weather.icon} alt="icon" />
                         <h4>wind: {weather.wind}mph</h4>
                        </>
                    )
                    
                })
                 
             }
 
            </>
        )
    }
    
    
}
 
export default DisplayCountry;
