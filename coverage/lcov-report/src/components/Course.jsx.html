import React from 'react';
import Header from './Header.jsx';
import Content from './Content.jsx';
import Total from './Total.jsx';
 
 
 
const Course = ({course}) => {
    console.log(course)
    return(
       <>
       <Header header = {course.name} />
       
       <ul>
         {
            course.parts.map(part => {
               return <Content key = {part.id} part = {part.name} numOfExcercises = {part.exercises}/>
            })
         }
       </ul>
 
       <Total total = {course.parts.map(part => part.exercises).reduce((sum, current) => {
           return sum + current;
       }, 0)} />
       </>
    )
}
 
export default Course;
